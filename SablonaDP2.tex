\section{Konstrukce univerza} \label{sec:konstrukce}
\subsection{Formalizace relace spl\v{n}ov\'{a}n\'{i}}
V t\'{e}to \v{c}\'{a}sti se budem v\v{e}novat konstrukci univerza $\mathbb{L}$ uvnit\v{r} teorie mno\v{z}in. Konstruktivn\'{i} universum, zde bude vytvo\v{r}eno pomocí formalizace logiky uvnit\v{r} teorie mno\v{z}in . 

Nejd\v{r}\'{i}ve si uka\v{z}eme, \v{z}e si vysta\v{c}\'{i}me jen se symboly pro konjunci($ \wedge $), negaci ($ \neg $) a existen\v{c}n\'{i} kvantifik\'{a}tor ($ \exists $).

\begin{veta}%\bfindex{Logick\'{a} ekvivalence} 
\textbf{Logick\'{a} ekvivalence \\ }
Nech\v{t} m\'{a}me formule $ \psi $ a $ \phi $. Pak plat\'{i} n\'{a}sleduj\'{i}c\'{i} ekvivalence:
\[ a) \psi \vee \phi  \equiv	\neg (\neg \psi \wedge \neg \phi) \]  
\[ b)\psi \rightarrow  \phi  \equiv 	\neg (\psi \wedge \neg \phi)  \]
\[ c) \forall x  \psi(x)  \equiv 	\neg \exists \neg  \psi(x)   \]
\textbf{D\r{u}kaz \\ }
a)  \[ \psi \vee \phi \Rightarrow \neg \psi \wedge \neg \phi \]  
sporem  
 \[ \psi \vee \phi   \curlywedge \neg \psi \wedge \neg \phi \] 
z druh\'{e} formule
\[   \neg \psi \curlywedge \neg \phi  \]
co\v{z} je ve sporu s t\'{i}m, \v{z}e mus\'{i} platit aspo\v{n} jedna z formul\'{i} 
\[  \psi \curlyvee \phi  . \]
Druh\'{a} \v{c}\'{a}st implikace 
 \[  \neg ( \neg \psi \wedge \neg \phi) \Rightarrow \psi \vee \phi  \]  
 obm\v{e}na  
\[   \neg (\psi \vee \phi) \Rightarrow \neg \psi \wedge \neg \phi \]  
Nech\v{t} tedy plat\'{i} p\v{r}edpoklad  
\[ \neg (\psi \vee \phi)  \]  
jinak je implikace trivi\'{a}ln\v{e} spln\v{e}na. Z toho dost\'{a}v\'{a}me 
\[ \neg \psi  \curlywedge  \neg \phi  \]
co\v{z} je p\v{r}esn\v{e} to, co jsme cht\v{e}li.\\

b) \[  \psi \rightarrow  \phi  \Rightarrow  	\neg (\psi \wedge \neg \phi) \]  
sporem  
\[ \psi \rightarrow  \phi \curlywedge 	\psi \wedge \neg \phi \]  
z 
\[ \psi \wedge \neg \phi  \] 
co\v{z} je ve sporu s t\'{i}m, \v{z}e plat\'{i} 
\[  \psi \rightarrow  \phi \]
Druh\'{a} \v{c}\'{a}st implikace 
\[ \neg (\psi \wedge \neg \phi)   \Rightarrow  \psi \rightarrow  \phi \]  
obm\v{e}na  
\[ \neg  (\psi \rightarrow \phi) \Rightarrow  (\psi \wedge \neg \phi) \]
 Nech\v{t} tedy plat\'{i} p\v{r}edpoklad 
 \[  \neg  (\psi \rightarrow \phi) \]  
  jinak je implikace trivi\'{a}ln\v{e} spln\v{e}na.  Z toho dost\'{a}v\'{a}me
\[ \psi \curlywedge \neg \phi \] 

c) \[ \forall x  \psi(x)  \Rightarrow	\neg \exists \neg  \psi(x) \] 
sporem 
\[ \forall x  \psi(x) \curlywedge \exists \neg  \psi(x) \]  
tedy existuje nejaky prvek strukutury pro kter\'{y} plat\'{i}  
\[ \psi(x) \curlywedge \neg  \psi(x) \] 
tedy spor
\[ \neg \exists \neg  \psi(x) \Rightarrow \forall x  \psi(x)  \] 
obm\v{e}na   
\[ \neg \forall x  \psi(x)  \Rightarrow	\exists \neg  \psi(x)  \]  
Nech\v{t} tedy plat\'{i} p\v{r}edpoklad  \[ \neg \forall x  \psi(x) \]  
 jinak je implikace trivi\'{a}ln\v{e} spln\v{e}na. Z toho dost\'{a}v\'{a}me,\v{z}e existuje prvek a struktury pro kter\'{y} platí 
 \[  \neg \psi(x)  \] 
 a z toho tedy plyne z\'{a}v\v{e}r. Q.E.D.
\end{veta} \label{vet:ekvivalence}
D\'{a}le si nadefinujeme a uk\'{a}\v{z}eme relativizaci formul\'{i} proto abychom ji pot\'{e} mohli pou\v{z}\'{i}t pro formalizaci relace spl\v{n}ov\'{a}n\'{i}.
\begin{definice}%\bfindex{Notace}
\textbf{Relativizace \\}
Nech\v{t} $ \mathcal{M} $ je t\v{r}\'{i}da, pak pro n\v{e}jakou formuli $ \phi $ v definujme indukc\'{i} $ \phi^{ \mathcal{M}} $ jako relativizaci $ \phi $ v $ \mathcal{M} $ \\
\[ a) (x=y)^{ \mathcal{M}} ~~~ jako ~~~ x=y  \]
\[ b) (x \in y)^{ \mathcal{M}} ~~~ jako  ~~~ x \in y  \]
\[ c) (\psi \wedge \lambda )^{ \mathcal{M}}  ~~~ jako ~~~  \psi^{ \mathcal{M}}  \wedge  \lambda^{ \mathcal{M}} \]
\[d)(\neg \psi)^{ \mathcal{M}} ~~~ jako ~~~ \neg (\psi)^{ \mathcal{M}}  \]
\[e) (\exists \psi)^{ \mathcal{M}} ~~~ jako ~~~ \exists x ( x \in \mathcal{M} \wedge (\psi)^{ \mathcal{M}} ) \] 
\label{def:relat}
\end{definice} 

Zavedeme si Quinovu rohovovou notaci a pomoc\'{i} n\'{i} si obohat\'{i}me teorii.

\begin{definice}%\bfindex{Notace}
\textbf{Quinova rohovov\'{a} notace \\ }
Nech\v{t} Ob n\v{e}jak\'{y} objekt v metateorii, tak $ \left\ulcorner  Ob\right\urcorner$ je konstanta, kter\'{a} zna\v{c}\'{i}, \v{z}e jsme teorii roz\v{s}\'{i}\v{r}ili o form\'{a}ln\'{i} definici Ob 
 \label{def:notace}
\end{definice}

\begin{lemma}
~\\
Pro libovolnou formuli $ \psi(x_0,..,x_{n-1}) $ \\
\[ \mathrm{ZF}\vdash \forall \langle M,\in \rangle  \forall x_0,..,x_{n-1} ( \psi^{M}(x_0,..,x_{n-1}) \Leftrightarrow \langle M,\in \rangle  \vDash \left\ulcorner \psi \right\urcorner[\langle x_0,..,x_{n-1})\rangle ]) \]
\textbf{D\r{u}kaz \\ }
Nech\v{t} m\'{a}me n\v{e}jak\'{y} model  $ \langle M,\in \rangle  $ a prom\v{e}nn\'{e} $ x_0,..,x_{n-1}$ :\\
D\r{u}kaz povedeme indukc\'{i} podle slo\v{z}itosti formule.\\
Ia)Tedy $ \psi(x_0,..,x_{n-1}) $ je atomick\'{a} formule tvaru x=y \\
podle relativizace $ (x=y)^{ \mathcal{M}} $ odpov\'{i}d\'{a} formule x=y,\\ 
kter\'{a} je v\v{s}ude interpretovan\'{a} v\v{s}ude jako relace identity, tedy i tady
\[ (x=y)^{M} \Leftrightarrow \langle M,\in \rangle  \vDash \left\ulcorner x=y \right\urcorner \]

Ib)Tedy $ \psi(x_0,..,x_{n-1}) $ je atomick\'{a} formule tvaru $ x \in y  $
podle relativizace $ (x \in y)^{ \mathcal{M}} $ odpov\'{i}d\'{a} formule $ x \in y  $, vzhledem k tomu,\v{z}e model $ \langle M,\in \rangle $ je uspo\v{r}\'{a}d\'{a}n pomoc\'{i} relace n\'{a}le\v{z}en\'{i} tak ur\v{c}it\v{e} plat\'{i}
\[ x \in y^{M} \Leftrightarrow \langle M,\in \rangle  \vDash \left\ulcorner x \in y \right\urcorner \]
t\'{i}m m\'{a}me dok\'{a}zano lemma pro atomick\'{e} formule.
\newpage
Nech\'{t} tedy m\'{a}me dok\'{a}z\'{a}no pro 
\[ \psi^{M}(x_0,..,x_{n-1}) \Leftrightarrow \langle M,\in \rangle  \vDash \left\ulcorner \psi \right\urcorner[\langle x_0,..,x_{n-1})\rangle ] \]
\[ \phi^{M}(x_0,..,x_{n-1}) \Leftrightarrow \langle M,\in \rangle  \vDash \left\ulcorner \phi \right\urcorner[\langle x_0,..,x_{n-1})\rangle ] \]
IIa)Tedy $ \pi(x_0,..\,x_{n-1}) $ je formule tvaru $ \psi(x_0,..\,x_{n-1}) \wedge \phi(x_0,..\,x_{n-1}) $\\
podle relativizace   \[ {(\psi(x_0,..\,x_{n-1}) \wedge \phi(x_0,..\,x_{n-1}))}^{M} \]  odpov\'{i}d\'{a}  \[ {\psi(x_0,..\,x_{n-1})^{M} \curlywedge \phi(x_0,..\,x_{n-1})}^{M}  \]
z induk\v{c}n\'{i}ho p\v{r}edpokladu dost\'{a}v\'{a}me 
\[ \langle M,\in \rangle  \vDash \left\ulcorner \phi \right\urcorner[\langle x_0,..,x_{n-1})\rangle ] \curlywedge \langle M,\in \rangle  \vDash \left\ulcorner \psi \right\urcorner[\langle x_0,..,x_{n-1})\rangle ]  \]
co\v{z} podle Tarsk\'{e}ho definice pravdy je pr\'{a}v\v{e}
\[\langle M,\in \rangle  \vDash \left\ulcorner \phi \wedge \psi \right\urcorner[\langle x_0,..,x_{n-1})\rangle ] \]
IIb)Tedy $ \pi(x_0,..,x_{n-1}) $ je formule tvaru $ \neg \psi(x_0,..,x_{n-1})  $
podle relativizace   
\[ {(\neg \psi(x_0,..,x_{n-1}))} ^{M} \]  odpov\'{i}d\'{a}  \[  \neg {(\psi(x_0,..,x_{n-1})}^{M}) \]
z induk\v{c}n\'{i}ho p\v{r}edpokladu dost\'{a}v\'{a}me 
\[\langle M,\in \rangle \nvDash \left\ulcorner \psi \right\urcorner[\langle x_0,..,x_{n-1})\rangle ]  \]
co\v{z} podle Tarsk\'{e}ho definice pravdy je pr\'{a}v\v{e}
\[\langle M,\in \rangle  \vDash \left\ulcorner \neg \psi \right\urcorner[\langle x_0,..,x_{n-1}\rangle ]  \]
IIc)Tedy $ \pi(x_0,..,x_{n-1}) $ je formule tvaru $ \exists \psi(x_0,..,x_{n-1})  $
podle relativizace   
\[ {(\exists  \psi(x_0,..,x_{n-1})) } ^{M} \]  odpov\'{i}d\'{a}  \[ \exists x ( x \in \mathcal{M} \wedge (\psi)^{ \mathcal{M}} ) \]
z induk\v{c}n\'{i}ho p\v{r}edpokladu dost\'{a}v\'{a}me 
\[\exists x ( x \in \mathcal{M} \curlywedge \langle M,\in \rangle  \vDash \left\ulcorner \psi \right\urcorner[\langle x_0,..,x_{n-1})\rangle ]) \]
co\v{z} podle Tarsk\'{e}ho definice pravdy je pr\'{a}v\v{e}
\[\langle M,\in \rangle  \vDash \left\ulcorner \exists \psi \right\urcorner[\langle x_0,..,x_{n-1}\rangle ]  \]
Q.E.D
\label{lem:formalizace}
\end{lemma}
\newpage
D\'{a}le si uk\'{a}\v{z}eme formalizaci definovateln\'{y}ch mno\v{z}in.
\begin{definice}%\bfindex{Notace}
\textbf{Ohodnocen\'{i} \\ }
Pro ka\v{z}d\'{e}  $\mathit{n} $ a n-tici $ \langle x_0,..,x_{n-1} \rangle $ $ \mathtt{s} $ je funkce s defi\v{c}n\'{i}m oborem n a hodnotami
$ \mathtt{s}(i)=x_i  $.
\end{definice}
\begin{definice}%\bfindex{Notace}
~\\
Nech\v{t} $ n \in \omega $ a i,j $ < $ n. \\
\[  Proj(A,R,n) =  \{ \mathtt{s} \in A^n : \exists t \in R (t \upharpoonright n = \mathtt{s}) \}  \]
\[  Diag_{\in}(A,n,i,j) =  \{ \mathtt{s} \in A^n : \mathtt{s}(i) \in \mathtt{s}(j) \} \] 
\[  Diag_{=}(A,n,i,j) =  \{ \mathtt{s} \in A^n : \mathtt{s}(i) = \mathtt{s}(j) \}   \]
Pot\'{e} rekurz\'{i} p\v{r}es  $ k \in \omega $ defimujeme  $ Df^*(k,A,n) $ : \\
$  a) Df^*(0,A,n) = \{ Diag_{\in}(A,n,i,j): i,j <  n \} \cup \{ Diag_{=}(A,n,i,j): i,j <  n \}  \ $
$  b) Df^*(k+1,A,n) = Df^*(k,A,n) \cup \{ A^n-R: R \in Df^*(k,A,n) \} \cup \newline \{ R \cap S: R,S \in Df^*(k,A,n) \}\cup \{ Proj(A,R,n) : R \in Df^*(k,A,n+1) \}    $
\\ z toho pak definujme
\[  Df(A,n)= \bigcup \{Df^*(k,A,n):k \in \omega \} \]  
\end{definice}
\begin{lemma}%\bfindex{Notace} 
~\\
Kdy\v{z} $ R \in Df(A,n)$,tak $A^n-R \in Df(A,n)$ \\
Kdy\v{z} $ R,S \in Df(A,n)$,tak $R \cap S \in Df(A,n)$ \\
Kdy\v{z} $ R \in Df(A,n+1)$,tak $Proj(A,R,n) \in Df(A,n)$ \\
\textbf{D\r{u}kaz \\ } 
Nech\v{t}  \[ R,S \in Df(A,n) \]
z definice Df(A,n) plyne, \v{z}e existuje $k_1$ a $k_2 $ tak \v{z}e 
\[ R \in Df^*(k_2,A,n)  \] 
\[ S \in Df^*(k_1,A,n) \]  
nech\v{t} tedy $x=max\{k_1,k_2\}$ tak 
\[ R \cap S \in Df^*(x+1,A,n) \] 
a tedy pak i 
\[ R \cap S \in Df(A,n) \] \\
\newpage
Nech\v{t}  \[ R \in Df(A,n) \] 
z definice Df(A,n) plyne, \v{z}e existuje k tak \v{z}e 
\[ R \in Df^*(k,A,n) \]
tedy z definice 
\[ A^n-R \in Df^*(x+1,A,n) \] 
a tedy pak i 
\[ A^n-R \in Df(A,n)  \]\\
Nech\v{t}  
\[ R \in Df(A,n+1) \] 
z definice Df(A,n+1) plyne, \v{z}e existuje k tak \v{z}e 
\[ R \in Df^*(k,A,n+1) \]
tedy z definice 
\[ Proj(A,R,n)\in Df^*(x+1,A,n) \] 
a tedy pak i 
\[ Proj(A,R,n) \in Df(A,n) \]
\label{lem:uzavr}
\end{lemma}

\begin{lemma}%\bfindex{Notace}
~\\
Nech\v{t} $ \phi(x_0,..,x_{n-1})  $je formule s voln\'{y}mi prom\v{e}n\'{y} $ x_0,...,x_{n-1}$ tak 
\[ \forall A [\{ \mathtt{s} \in A^n: \phi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)]   \]
\textbf{D\r{u}kaz \\ } 
Indukc\'{i} podle slo\v{z}itosti formule s fixovanym A. \\
Nech\v{t} $ \phi $ je atomatick\'{a} formule $ x_i \in x_j $ pak z 
\[ Diag_{\in}(A,n,i,j) \in Df(A,n) \]  
plyne 
\[ [\{ \mathtt{s} \in A^n: \phi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \]
Stejn\v{e} tak pro $ x_i = x_j $ pak z  
\[ Diag_{=}(A,n,i,j) \in Df(A,n)  \] 
plyne 
\[ [\{ \mathtt{s} \in A^n: \phi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \]
Nech\v{t} $ \phi $ je $ \psi \wedge \chi $ a v\'{i}me z induk\v{c}n\'{i}ho p\v{r}edpokladu , \v{z}e plat\'{i} 
\[ [\{ \mathtt{s} \in A^n: \psi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \]
\[ [\{ \mathtt{s} \in A^n: \chi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \] 
a tedy podle p\v{r}edchoz\'{i}ho lemmatu tedy plat\'{i} 
\[ [\{ \mathtt{s} \in A^n: \chi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \cap  \{ \mathtt{s} \in A^n: \psi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \] 
co\v{z} je p\v{r}esn\v{e}
\[ [\{ \mathtt{s} \in A^n: (\chi \wedge \psi)^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \]  
Nech\v{t} $ \phi $ je  $ \neg \psi $   a v\'{i}me z induk\v{c}n\'{i}ho p\v{r}edpokladu , \v{z}e plat\'{i} 
\[ [\{ \mathtt{s} \in A^n: \psi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \] 
a tedy podle p\v{r}edchoz\'{i}ho lemmatu tedy plat{i} 
\[ [A-\{ \mathtt{s} \in A^n: \psi^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \] 
co\v{z} je p\v{r}esn\v{e}
\[ [\{ \mathtt{s} \in A^n: (\neg  \psi)^\mathrm{A} (\mathtt{s}(0),..,\mathtt{s}(n-1)\} \in Df(A,n)] \] .
Kone\v{c}n\v{e}  nech\v{t} $ \phi $ je  $ \exists y ~\psi $ \\

Pro ka\v{z}dou existen\v{c}n\'{i} formuli $ \phi_{i}(x_1,...,x_i) $ tvaru $ \exists a \phi_{j}(x_1,...,x_i,a) $ ze seznamu plat\'{i}:
\[ \forall x_1,...,x_i \in A (\phi_{i}^\mathrm{B}(x_1,..x_n) \rightarrow \exists a \in A \phi_{j}^\mathrm{B}(x_1,..x_n,a) ) (2) \]
\textbf{D\r{u}kaz \\ }
Nech\v{t} tedy m\'{a}me n\v{e}jak\'{e} $ x_1,...,x_i \in A  $ a nech\v{t} 
\[ \forall x_1,...,x_i \in A (\phi_{t}^\mathrm{A}(x_1,..x_n) \leftrightarrow  \phi_{t}^\mathrm{B}(x_1,..x_n) )  \] 
a p\v{r}edpokl\'{a}dejme
\[ \phi_{i}^\mathrm{B}(x_1,..x_n)  \]
z p\v{r}edpokladu pro  $ \phi_{i} $ plat\'{i}
\[ \phi_{i}^\mathrm{A}(x_1,..x_n)  \]
to je podle definice a relativizace
\[ (\exists a \in A) (\phi_{j}^\mathrm{A}(x_1,..x_n,a))  \]
z p\v{r}edpokladu pro  $ \phi_{j} $ plat\'{i}
\[ (\exists a \in A) (\phi_{j}^\mathrm{B}(x_1,..x_n,a))  \]
\newpage
druhou implikaci dok\'{a}\v{z}eme indukc\'{i} podle slo\v{z}itosti formule \\
pro v\v{s}echny formule $  \psi_{i} $ bez kvantifikatoru plat\'{i} (1) z lemma \ref{lem:transa}\\
Nech\v{t} tedy  $  \psi_{i} $ je $ \exists a \psi_{j}(x_1,..x_n,a) $ m\'{a}me n\v{e}jak\'{e} $ x_1,...,x_i \in A  $ \\
 z definice relativizace 
\[  (\phi_{i}^\mathrm{A}(x_1,..x_n))  \Leftrightarrow \exists a \in A (\phi_{j}^\mathrm{A}(x_1,..x_n,a)) \]
z induk\v{c}n\'{i}ho p\v{r}edpokladu 
\[ \exists a \in A (\phi_{j}^\mathrm{A}(x_1,..x_n,a)) \Leftrightarrow \exists a \in A (\phi_{j}^\mathrm{B}(x_1,..x_n,a)) \]
z $  A \subset B $ plyne 
\[ \exists a \in A (\phi_{j}^\mathrm{B}(x_1,..x_n,a)) \Rightarrow \exists a \in B (\phi_{j}^\mathrm{B}(x_1,..x_n,a))
 \]
z definice relativizace 
\[  \exists a \in B (\phi_{j}^\mathrm{B}(x_1,..x_n,a)) \Leftrightarrow   (\phi_{i}^\mathrm{B}(x_1,..x_n))   \]
implikace 
\[  (\phi_{i}^\mathrm{B}(x_1,..x_n))  \Rightarrow \exists a \in A (\phi_{j}^\mathrm{B}(x_1,..x_n,a))  \]
z p\v{r}edpokladu.
\label{lem:ref}
\end{lemma}
